## 使用示例

### 采集直辖市

```python
from scripts.fetcher import GovDocFetcher

fetcher = GovDocFetcher()

# 北京市
items = fetcher.fetch_list('beijing', 'provincial')
print(f"北京市: {len(items)} 条")

# 上海市
items = fetcher.fetch_list('shanghai', 'provincial')
print(f"上海市: {len(items)} 条")
```

### 采集多个省份

```python
from scripts.fetcher import GovDocFetcher

fetcher = GovDocFetcher()

provinces = ['guangdong', 'jiangsu', 'zhejiang', 'shandong']
for key in provinces:
    try:
        items = fetcher.fetch_list(key, 'provincial')
        print(f"{key}: {len(items)} 条")
    except Exception as e:
        print(f"{key}: 失败")
```

### 全国汇总

```python
from scripts.fetcher import GovDocFetcher
import json

fetcher = GovDocFetcher()
config_file = fetcher.config_dir / "provincial.json"

with open(config_file, 'r', encoding='utf-8') as f:
    all_sites = json.load(f)

all_items = []
for key in all_sites.keys():
    try:
        items = fetcher.fetch_list(key, 'provincial')
        for item in items:
            item['province'] = key
            all_items.append(item)
    except:
        pass

print(f"全国省级政策总计: {len(all_items)} 条")
```
